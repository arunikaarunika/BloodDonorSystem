CREATE TABLE DONOR_REG_TBL (
    REGISTRATION_ID            VARCHAR2(10) PRIMARY KEY,
    CAMP_DATE                  DATE,
    DONOR_NAME                 VARCHAR2(150),
    AGE_YEARS                  NUMBER(3),
    GENDER                     VARCHAR2(10),
    CONTACT_MOBILE             VARCHAR2(20),
    EMAIL                      VARCHAR2(120),
    BLOOD_GROUP                VARCHAR2(10),
    SELF_DECL_RECENT_ILLNESS   VARCHAR2(5),
    SELF_DECL_MEDICATION       VARCHAR2(5),
    PREFERRED_ARRIVAL_SLOT     VARCHAR2(30),
    REGISTRATION_STATUS        VARCHAR2(25),
    CREATED_TIMESTAMP          DATE,
    CANCELLATION_REASON        VARCHAR2(400)
);
